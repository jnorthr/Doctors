// filename: RunAsciidoc.groovy +
@Grab('org.asciidoctor:asciidoctor-java-integration:0.1.3') +
import org.asciidoctor.* +
def asciidoctor = Asciidoctor.Factory.create() +
def output = asciidoctor.renderFile(new File(args[0]),  [:]) +
println(output); +
